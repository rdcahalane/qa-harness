Global Testing Rules

After every commit + push

  • Open the deployed app in a Playwright headless browser and verify the changed UI path
  • Test actual UI interactions — not just API endpoints
  • For CDI: run test-teams-scenarios.mjs (11 scenarios) + test-bot.js
  • For ops-maturity: verify admin/users page, assessment create flow, and PPTX export
  • For lns-geo: verify map canvas renders, controls panel loads, scroll isolation works
  • Send iMessage to +14407493614 if blocked waiting for input or when deploy confirmed
  • Never declare a visual bug fixed without Playwright screenshot verification

Playwright is at /tmp/node_modules/playwright

  • Installed globally in /tmp — always const { chromium } = require('playwright')
  • Run from cd /tmp before invoking playwright scripts
  • Screenshots save to /tmp/*.png for inspection
  • Standard viewport: 1280×800 (desktop) or 900×650 (map views)

Ops Maturity

Ops Maturity
▶ Playwright — post-deploy UI verification Standard pattern: verify admin, assessment, and export flows Ops Maturity

Standard post-deploy check

cd /tmp && node -e "
const { chromium } = require('playwright');
(async () => {
  const browser = await chromium.launch({ headless: true });
  const page = await browser.newPage();
  await page.setViewportSize({ width: 1280, height: 800 });

  // Sign in
  await page.goto('https://ops-maturity.vercel.app/auth/signin', { waitUntil: 'networkidle', timeout: 30000 });
  await page.fill('input[name=\"email\"]', 'YOUR_EMAIL');
  await page.fill('input[name=\"password\"]', 'YOUR_PASSWORD');
  await Promise.all([page.waitForNavigation({ timeout: 15000 }), page.click('button[type=\"submit\"]')]);
  await new Promise(r => setTimeout(r, 2000));

  // Test admin/users
  await page.goto('https://ops-maturity.vercel.app/admin/users', { waitUntil: 'networkidle', timeout: 30000 });
  const body = await page.textContent('body');
  console.log('App error?', body.includes('Application error'));
  console.log('Users page loaded?', page.url().includes('admin/users'));

  await page.screenshot({ path: '/tmp/om-admin-users.png' });
  await browser.close();
})();
"

lns-geo (WorldView Map)

lns-geo
▶ Playwright — map and controls verification Verify map renders, controls panel loads, scroll isolation works lns-geo

Standard post-deploy check

cd /tmp && node -e "
const { chromium } = require('playwright');
(async () => {
  const browser = await chromium.launch({ headless: true });
  const page = await browser.newPage();
  await page.setViewportSize({ width: 900, height: 650 });
  await page.goto('https://lns-geo.pages.dev', { waitUntil: 'networkidle', timeout: 30000 });
  await new Promise(r => setTimeout(r, 4000));

  // Map canvas
  const canvas = await page.\$('canvas');
  const box = await canvas?.boundingBox();
  console.log('Map canvas:', box?.width > 0 ? 'OK' : 'MISSING');

  // Controls panel
  const panel = await page.\$('[class*=\"rounded-xl\"]');
  console.log('Controls panel:', panel ? 'OK' : 'MISSING');

  // Scroll isolation
  const before = await page.evaluate(() => document.querySelector('[style*=\"overflow\"]')?.scrollTop ?? 0);
  await page.mouse.wheel(0, 200);
  await new Promise(r => setTimeout(r, 500));
  const after = await page.evaluate(() => document.querySelector('[style*=\"overflow\"]')?.scrollTop ?? 0);
  console.log('Scroll isolation:', after > before ? 'OK' : 'CHECK');

  await page.screenshot({ path: '/tmp/lns-geo.png' });
  await browser.close();
})();
"

Checks

Map canvas renders Controls panel visible Scroll isolation (no map zoom) Signals tab loads Filters tab loads

▶ Mobile screenshot sweep — all platforms 390×844 + 768×1024 for CDI, OM (authed), Genome, lns-geo

Full sweep (requires OM credentials in OpenBrain)

cd /tmp && node -e "
const { chromium } = require('playwright');
(async () => {
  const browser = await chromium.launch({ headless: true });
  const viewports = [
    { name: 'mobile', width: 390,  height: 844  },
    { name: 'tablet', width: 768,  height: 1024 },
  ];

  // ── Unauthenticated sites ──────────────────────
  const publicSites = [
    { name: 'genome',  url: 'https://genome-api-production.up.railway.app' },
    { name: 'cdi',     url: 'https://cdi.sera-holt.workers.dev' },
    { name: 'lns-geo', url: 'https://lns-geo.pages.dev' },
  ];
  for (const vp of viewports) {
    for (const site of publicSites) {
      const page = await browser.newPage();
      await page.setViewportSize({ width: vp.width, height: vp.height });
      await page.goto(site.url, { waitUntil: 'networkidle', timeout: 30000 });
      await new Promise(r => setTimeout(r, 3000));
      await page.screenshot({ path: '/tmp/' + site.name + '-' + vp.name + '.png' });
      const body = await page.textContent('body');
      console.log(site.name, vp.name, '— error?', body.includes('Application error'));
      await page.close();
    }
  }

  // ── Ops Maturity (authenticated) ──────────────
  for (const vp of viewports) {
    const page = await browser.newPage();
    await page.setViewportSize({ width: vp.width, height: vp.height });
    await page.goto('https://ops-maturity.vercel.app/auth/signin', { waitUntil: 'networkidle', timeout: 30000 });
    await page.fill('input[name=\"email\"]', 'rdcahalane@gmail.com');
    await page.fill('input[name=\"password\"]', 'Axiom2026!QA');
    await Promise.all([page.waitForNavigation({ timeout: 15000 }), page.click('button[type=\"submit\"]')]);
    await new Promise(r => setTimeout(r, 2000));
    await page.screenshot({ path: '/tmp/om-home-' + vp.name + '.png' });
    await page.goto('https://ops-maturity.vercel.app/admin/users', { waitUntil: 'networkidle', timeout: 20000 });
    await new Promise(r => setTimeout(r, 2000));
    await page.screenshot({ path: '/tmp/om-admin-' + vp.name + '.png' });
    console.log('om', vp.name, 'done — url:', page.url());
    await page.close();
  }

  await browser.close();
  console.log('All screenshots saved to /tmp/');
})();
"
